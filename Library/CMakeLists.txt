	DiscordCoreAPI PUBLIC
	"$<$<NOT:$<STREQUAL:$<INSTALL_INTERFACE:$<INSTALL_PREFIX>/include>,${VCPKG_INSTALLED_DIR}/include>>:$<INSTALL_INTERFACE:$<INSTALL_PREFIX>/include>>"

	IMPORTED_CONFIGURATIONS "Release;Debug"
